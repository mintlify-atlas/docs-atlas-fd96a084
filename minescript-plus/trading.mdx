The `Trading` class provides methods for interacting with villager and merchant trades.

<Note>
  All Trading methods require an active trading screen to be open. Open a trading GUI by right-clicking a villager or wandering trader.
</Note>

## get_offers

Retrieves all merchant offers available in the current trading screen.

**Returns:** `MerchantOffers` - The offers object, or None if no trading screen is open.

**Source:** `minescript_plus.py:1325`

```python
import minescript_plus as ms

# Get all available offers
offers = ms.Trading.get_offers()
if offers:
    count = offers.size()
    print(f"Available trades: {count}")
```

## get_costA

Gets the first cost item (costA) for a merchant offer.

<ParamField path="offer_index" type="int" required>
  The offer index.
</ParamField>

<ParamField path="name_and_count" type="bool" default="False">
  If True, returns (name, count) tuple. Otherwise returns ItemStack.
</ParamField>

**Returns:** `tuple[str, int] | ItemStack | None` - ItemStack or (name, count) tuple, or None if not available.

**Source:** `minescript_plus.py:1356`

```python
import minescript_plus as ms

# Get cost as tuple
cost = ms.Trading.get_costA(0, name_and_count=True)
if cost:
    name, count = cost
    print(f"Cost A: {count}x {name}")

# Get cost as ItemStack
item_stack = ms.Trading.get_costA(0)
```

</ParamField>

## Complete Example: Trade Inspector

```python
import minescript_plus as ms
import time

def inspect_all_trades():
    """Display all available trades from a merchant."""
    
    # Wait for trading screen to open
    print("Open a trading GUI...")
    time.sleep(2)
    
    offers = ms.Trading.get_offers()
    if not offers:
        print("No trading screen open!")
        return
    
    count = offers.size()
    print(f"\n=== Available Trades ({count}) ===")
    
    for i in range(count):
        print(f"\nTrade #{i + 1}:")
        
        # Get costs
        cost_a = ms.Trading.get_costA(i, name_and_count=True)
        if cost_a:
            name, qty = cost_a
            print(f"  Cost A: {qty}x {name}")
        
        cost_b = ms.Trading.get_costB(i, name_and_count=True)
        if cost_b:
            name, qty = cost_b
            print(f"  Cost B: {qty}x {name}")
        
        # Get result
        result = ms.Trading.get_result(i, name_and_count=True)
        if result:
            name, qty = result
            print(f"  Result: {qty}x {name}")
        
        # Check if trade is maxed out
        offer = ms.Trading.get_offer(i)
        if offer and offer.isOutOfStock():
            print("  ⚠️ OUT OF STOCK")

inspect_all_trades()
```

## Auto-Trader Example

```python
import minescript_plus as ms
import time

def auto_trade(offer_index: int, times: int = 1):
    """Automatically execute a specific trade multiple times."""
    
    offers = ms.Trading.get_offers()
    if not offers:
        print("No trading screen open!")
        return
    
    # Validate offer exists
    offer = ms.Trading.get_offer(offer_index)
    if not offer:
        print(f"Trade #{offer_index} doesn't exist!")
        return
    
    # Display what we're trading
    cost_a = ms.Trading.get_costA(offer_index, name_and_count=True)
    result = ms.Trading.get_result(offer_index, name_and_count=True)
    
    print(f"Trading {cost_a[1]}x {cost_a[0]} for {result[1]}x {result[0]}")
    print(f"Executing {times} time(s)...")
    
    for i in range(times):
        # Check if still in stock
        if offer.isOutOfStock():
            print(f"Trade out of stock after {i} trades!")
            break
        
        ms.Trading.trade_offer(offer_index)
        print(f"  Trade {i + 1}/{times} completed")
        time.sleep(0.3)  # Small delay between trades
    
    print("Trading complete!")

# Trade the first offer 10 times
auto_trade(0, times=10)
```

## Find Best Emerald Trade

```python
import minescript_plus as ms

def find_best_emerald_trade():
    """Find the trade that gives the most emeralds."""
    
    offers = ms.Trading.get_offers()
    if not offers:
        print("No trading screen open!")
        return None
    
    best_trade = None
    best_count = 0
    
    for i in range(offers.size()):
        result = ms.Trading.get_result(i, name_and_count=True)
        
        if result:
            name, count = result
            if "emerald" in name.lower() and count > best_count:
                best_count = count
                best_trade = i
    
    if best_trade is not None:
        cost_a = ms.Trading.get_costA(best_trade, name_and_count=True)
        print(f"Best emerald trade: #{best_trade + 1}")
        print(f"  Give: {cost_a[1]}x {cost_a[0]}")
        print(f"  Get: {best_count}x Emerald")
        return best_trade
    else:
        print("No emerald trades found")
        return None

best = find_best_emerald_trade()
if best is not None:
    # Execute the best trade
    ms.Trading.trade_offer(best)
```
